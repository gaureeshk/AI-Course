\section*{Problem GATE-EC-2019-Q47: }
Let $X$ and $Y$ be jointly distributed random
variables having the joint probability
density function
\[
f(x,y) = \begin{cases}
            \frac{1}{\pi}, &\text{if}\quad x^2 + y^2 \leq 1\\
             0, &\text{otherwise}\\
            \end{cases}
\]
Then $P(Y > \text{max}(X,-X))$ is

\section*{Solution:}
Since we require $Y > \text{max}(X,-X)$
\begin{align}
    \text{max}(X,-X) = |X|
\end{align}
We need to find
\begin{align}
    \pr{Y>|X|}
\end{align}
Integrating the probability distribution function on our domain
\begin{align}
\pr{Y > |X|} &= \iint_{\{y > \left|x\right|\}} f(x,y) dx\,dy\\
&= \iint_{\{y > \left|x\right|, \ x^2 + y^2 \leq 1\}}  \frac{1}{\pi} dx\,dy\\
&\quad + \iint_{\{y > \left|x\right|, \ x^2 + y^2 > 1\}}  0dx\,dy\\
&= \frac{1}{\pi} \iint_{\{y > \left|x\right|, \ x^2 + y^2 \leq 1\}} dx\,dy
\end{align}
On converting to $r$ and $\theta$ form, we get:
\begin{align}
\pr{Y > |X|} &= \frac{1}{\pi} \int_{\pi/4}^{3\pi/4} \int_0^1 r\,dr\,d\theta \\
&= \frac{1}{\pi} \int_{\pi/4}^{3\pi/4} \left(\frac{1}{2}r^2\right)\bigg|_0^1 d\theta\\
&= \frac{1}{\pi} \int_{\pi/4}^{3\pi/4} \frac{1}{2} d\theta\\
&= \frac{1}{2\pi} \left(\frac{3\pi}{4} - \frac{\pi}{4}\right)\\
&= \frac{1}{2\pi} \cdot \frac{\pi}{2}\\
&= \frac{1}{4}
\end{align}
